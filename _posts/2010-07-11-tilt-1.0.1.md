## Installation

    > gem install tilt

## Configuration

    # in Gemfile:
    gem 'erb' # or 'haml' or 'liquid' or ...
    gem 'tilt'
    
    # in an initializer:
    require 'erb'
    require 'tilt'

## Usage

    # engine determined automatically by file extension:
    template = Tilt.new('templates/foo.erb')
    output = template.render :x => 22, :y => nil
